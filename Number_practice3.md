```js
// find the square root of 25

console.log('SQRT of 25 \n' ,Math.sqrt(25)) 

// find the value of 3 to the power of 5

console.log(Math.pow(3, 5))

// find the value of 7.4 rounded

console.log('7.4 rounded \n' ,Math.round(7.4))

// find the rounded-up value of 7.4

console.log('7.4 rounded-up \n' ,Math.round(7.4))

// find the absolute value of -13

console.log('Absolute value of -13', '\n',Math.abs(-13))

// find the minimum value in a set of numbers: 12,13,14,15,2,-1

console.log('Minimum valuse from set 12,13,14,15,2,-1', '\n', Math.min(12,13,14,15,2,-1))

// find the maximum value in a set of numbers: 12,13,14,15,2,-1

console.log(Math.max(12,13,14,15,2,-1))


// create a random number

console. log('Random number between 0 and 10', '\n', Math.trunc(Math.random))

```
